```js
var price = 100;
function doSomething() {
    var doublePrice = price * 2;
    var price = 200;
    var halfPrice = price / 2;
    console.log(doublePrice);
    console.log(halfPrice);
    console.log(price, 'start pirce')
}
doSomething();
console.log(price, 'end price')
```

NaN

100

200 'start price'
